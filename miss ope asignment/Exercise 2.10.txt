 a) System.out.printf("x = %d%n", x + 5); 
 b) System.out.printf("Value of %d *%d is %d\n", x, y, (x * y) ); 
 c) System.out.printf("x is %d and y is %d", x, y); 
 d) System.out.printf("%d is not equal to %d\n", (x + y), (x * y) );
                   Solution
